```sql
WITH CTE_F AS
(
SELECT 
requester_id as one,
accepter_id as two
FROM request_accepted
UNION 
SELECT 
accepter_id,
requester_id
FROM request_accepted
  ),
CTE_F2 AS
(
SELECT 
one,
count(two) as ff,
DENSE_RANK() over ( order by COUNT(two) DESC) as r
FROM
CTE_F
group by 
one
)
SELECT one, ff
from CTE_F2
WHERE r=1
```
